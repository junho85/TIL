# 3. Longest Substring Without Repeating Characters

## 풀이
* 입력되는 문자에서 각 자리부터 시작하는 중복되지 않은 substring 구해서 길이를 구함
* set 을 이용해서 처음부터 훑으면서 중복 되는 문자가 나올 때 까지 substring 을 찾는다.
* 그 길이를 max 비교
* 156 ms, 10.56%

package leetcode;

import org.junit.Test;

import java.util.HashSet;
import java.util.Set;

import static org.assertj.core.api.Assertions.assertThat;

public class PS3LongestSubstringWithoutRepeatingCharacters {

    class Solution {
        Long getLengthOfNotRepeatingCharacterSubstring(String s) {
            Set<Character> tmp = new HashSet<>();
            for (int i=0; i<s.length(); i++) {
                if (!tmp.add(s.charAt(i))) {
                    return (long) i;
                }
            }
            return (long) s.length();
        }

        public int lengthOfLongestSubstring(String s) {
            if (s.length() == 0) {
                return 0;
            }

            long max = 0;
            for (int i=0; i<s.length(); i++) {
                long ret = getLengthOfNotRepeatingCharacterSubstring(s.substring(i));
                if (ret > max) {
                    max = ret;
                }
                if (s.length() - i < max) {
                    return (int)max;
                }
            }

            return (int)max;
        }
    }
