# 3. Longest Substring Without Repeating Characters

## 문제
* 문자가 겹치지 않은 substring 중에서 가장 긴 것의 길이를 구하는 것.

## 가장 긴 substring 에서 문자가 겹치지 않은것을 찾기
* timeout 걸려서 실패

## 입력되는 문자에서 각 자리부터 시작하는 중복되지 않은 substring 구해서 길이를 구함
* set 을 이용해서 처음부터 훑으면서 중복 되는 문자가 나올 때 까지 substring 을 찾는다.
* 그 길이를 max 비교
* 성공은 하지만 경우에 따라 빠르지는 않음.

```cpp
#include <string>
#include <cassert>
#include <iostream>
#include <vector>
#include <set>

using namespace std;

class Solution {
public:
    unsigned long get_length_of_not_repeating_character_substring(string s) {
        set<char> tmp;
        for (unsigned long i=0; i<s.size(); i++) {
//            cout << s[i] << endl;
            auto ret = tmp.insert(s[i]);
            if (ret.second == false) {
                return i;
            }
        }
        return s.size();
    }

    int lengthOfLongestSubstring(string s) {
        if (s.size() == 0) {
            return 0;
        }

        // Runtime: 428ms, faster than 2.88% of C++ online submissions.
        unsigned long max = 0;
        for (int i=0; i<s.size(); i++) {
            unsigned long ret = get_length_of_not_repeating_character_substring(s.substr(i));
            if (ret > max) {
                max = ret;
            }
            if (s.size() - i < max) {
                return (int)max;
            }
        }

        return (int)max;
    }
};
```
